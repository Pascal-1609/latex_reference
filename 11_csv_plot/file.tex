\documentclass{article}

\usepackage{siunitx}
% to generate the plot from csv
\usepackage{tikz}
\usepackage{pgfplots}

% allows to place the legend below plot
\pgfplotsset{compat=newest}
% allows to enter the units nicely
\usepgfplotslibrary{units}

\sisetup{
  round-mode          = places,
  round-precision     = 2,
}

\begin{document}

	\begin{figure}[h!]
	\begin{center}
		\begin{tikzpicture}
		\begin{axis}[
			% scale the plot to \linewidth
			width=\linewidth,
			% display a grid
			grid=major,
			% set the style
			grid style={dashed,gray!30},
			% set the labels
			xlabel=X Axis $U$,
			ylabel=Y Axis $I$,
			% set the respective units
			x unit=\si{\volt},
			y unit=\si{\ampere},
			% put the legend below the plot
			legend style={at={(0.5,-0.2)},anchor=north},
			% display labels sideways
			x tick label style={rotate=90,anchor=east}
		]
			\addplot 
			% add a plot from table; you select the columns by using the actual name in
			% the .csv file (on top)
			table[x=column 1,y=column 2,col sep=comma] {table.csv}; 
			\legend{Plot}
		\end{axis}
		\end{tikzpicture}
		\caption{My first autogenerated plot.}
	\end{center}
	\end{figure}

\end{document}
