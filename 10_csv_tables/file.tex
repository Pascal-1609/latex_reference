\documentclass{article}

% for \toprule, \midrule and \bottomrule
\usepackage{booktabs}
% formats the units and values
\usepackage{siunitx}
% generates table from .csv
\usepackage{pgfplotstable}

% setup siunitx:
\sisetup{
	% rounds numbers
	round-mode          = places,
	% to 2 places
	round-precision     = 2,
}

\begin{document}

	\begin{table}[h!]
	\begin{center}
		\caption{Autogenerated table from .csv file.}
		\label{table1}
		\pgfplotstabletypeset[
			% allows to have multicolumn names
			multicolumn names,
			% the seperator in the .csv file
			col sep=comma,
			display columns/0/.style={
				% name of first column
				column name=$Value 1$,
				% use siunitx for formatting
				column type={S},string type
			},
			display columns/1/.style={
				column name=$Value 2$,
				column type={S},string type
			},
			every head row/.style={
				% have a rule at top
				before row={\toprule},
				after row={
					% the units seperated by &
					\si{\ampere} & \si{\volt}\\
					% rule under units
					\midrule
				}
			},
			% rule at bottom
			every last row/.style={after row=\bottomrule},
		% filename/path to file
		]{table.csv}
	\end{center}
	\end{table}

\end{document}

% this only works with tables that fit on one page
